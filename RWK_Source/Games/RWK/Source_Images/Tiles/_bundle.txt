{

	Bundle_[BUNDLE]()	{gBundle_[BUNDLE]=this;}
	void			Load();

	Array<Sprite*>		mTile_Paint;

[begin RAPT.CPP];
#include "Bundle_[BUNDLE].h"
Bundle_[BUNDLE]* gBundle_[BUNDLE]=NULL;

void Bundle_[BUNDLE]::Load()
{
	OverrideTextureSize();
	if (IsRetina("[BUNDLE]")) {if (!SpriteBundle::Load("[BUNDLE]@2X")) return;LoadData("[BUNDLE]@2X");}
	else {if (!SpriteBundle::Load("[BUNDLE]")) return;LoadData("[BUNDLE]");}
	
[AUTOMATIC]

	for (int aTileType=0;aTileType<10;aTileType++)
	{
		for (int aTile=0;aTile<mPaint[aTileType].Size();aTile++)
		{
			mTile_Paint.Add(&mPaint[aTileType][aTile]);
		}
	}

	gG.RegisterFillrect(&mFillrect);
	UnloadData();
}
[end];
